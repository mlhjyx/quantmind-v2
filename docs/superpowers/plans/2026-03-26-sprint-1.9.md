```python
def _get_trading_days(self) -> list[date]:
    """从DB加载交易日历。"""
    result = pd.read_sql(
        """SELECT cal_date FROM trading_calendar
           WHERE exchange = 'SSE' AND is_open = 1
             AND cal_date BETWEEN %s AND %s
           ORDER BY cal_date""",
        self.conn,
        params=(self.start_date, self.end_date + timedelta(days=60)),
    )
    assert isinstance(result, pd.DataFrame)
    return [d.date() if hasattr(d, 'date') else d for d in result["cal_date"].tolist()]

def _calc_ic_decay(self, factor_df: pd.DataFrame) -> dict[int, float]:
    """计算IC衰减——使用交易日偏移。"""
    assert self._excess_returns is not None

    # 加载交易日历（懒缓存）
    if not hasattr(self, '_trading_days_cache') or self._trading_days_cache is None:
        self._trading_days_cache = self._get_trading_days()
    td_list = self._trading_days_cache
    td_set = set(td_list)

    decay: dict[int, float] = {}
    dates = sorted(factor_df["trade_date"].unique())
    sampled_dates = dates[:: self.sample_step]

    for h in self.horizons:
        ic_values: list[float] = []

        for dt in sampled_dates:
            cross_section = factor_df[factor_df["trade_date"] == dt]

            # 用交易日历找到T+h个交易日后的日期
            try:

            except ValueError:

            end_idx = idx + h
            if end_idx >= len(td_list):
                continue
            future_end = td_list[end_idx]

            future_rets = self._excess_returns[
                (self._excess_returns["trade_date"] > dt)
                & (self._excess_returns["trade_date"] <= future_end)
            ]
            if future_rets.empty:
                continue

            cum_ret = pd.Series(future_rets.groupby("code")["excess_ret"].sum())
            fwd_ret = cum_ret.rename("fwd_ret")
            merged = cross_section.set_index("code")["neutral_value"].to_frame()
            merged = merged.join(fwd_ret, how="inner").dropna()

            if len(merged) < 30:
                continue

            result = stats.spearmanr(merged["neutral_value"], merged["fwd_ret"])
            ic_val = float(result.statistic)  # type: ignore[union-attr]

                ic_values.append(ic_val)

        decay[h] = float(np.mean(ic_values)) if ic_values else 0.0
```

```python
HEALTH_CHECK_KEY = "task_status:{date}:health_check"


def _check_health_prerequisite(trade_date: date, redis_client=None) -> bool:
    """检查健康预检是否通过。

    Args:
        trade_date: 交易日期。
        redis_client: Redis客户端（可注入用于测试）。

    Returns:
        True=可以继续, False=健康检查未通过应跳过。
    """
    if redis_client is None:
        from app.tasks.celery_app import celery_app as _app
        redis_client = _app.backend.client

    key = HEALTH_CHECK_KEY.format(date=trade_date.isoformat())
    raw = redis_client.get(key)

    if raw is None:
        logger.warning("[TaskDep] 无健康检查结果(key=%s)，首次运行放行", key)
        return True

    try:
        result = json.loads(raw)
        if not result.get("all_pass", False):
            logger.error(
                "[TaskDep] 健康检查未通过: %s，跳过信号生成",
                {k: v for k, v in result.items() if v is False},
            )
            return False
        return True
    except (json.JSONDecodeError, TypeError) as e:
        logger.error("[TaskDep] 健康检查结果解析失败: %s", e)
        return True  # 解析失败放行，避免假阳性阻塞
```

```python
"""base_broker.py — Broker策略模式抽象基类。

CLAUDE.md设计决策: Paper/实盘/外汇共用同一套因子→信号→风控链路，
唯一区别是执行层。用策略模式切换。

三个实现:
- SimBroker (回测, backtest_engine.py)
- PaperBroker (Paper Trading, paper_broker.py)
- MiniQMTBroker (A股实盘, broker_qmt.py)
"""

from abc import ABC, abstractmethod
from dataclasses import dataclass
from datetime import date
from typing import Optional


@dataclass
class Position:
    """统一持仓表示。"""
    code: str
    shares: int
    avg_cost: float = 0.0
    market_value: float = 0.0


class BaseBroker(ABC):
    """Broker抽象基类。所有执行层实现必须继承。"""

    @abstractmethod
    def get_positions(self) -> dict[str, int]:
        """获取当前持仓。Returns: {code: shares}"""
        ...

    @abstractmethod
    def get_cash(self) -> float:
        """获取可用现金。"""
        ...

    @abstractmethod
    def get_total_value(self, prices: Optional[dict[str, float]] = None) -> float:
        """获取总资产(现金+持仓市值)。"""
        ...


def get_broker(mode: str, **kwargs) -> BaseBroker:
    """工厂函数: 根据execution_mode创建Broker。

    Args:
        mode: 'paper' / 'live' / 'backtest'
        **kwargs: 传递给具体Broker构造函数。

    Returns:
        BaseBroker实例。
    """
    if mode == "paper":
        from engines.paper_broker import PaperBroker
        return PaperBroker(**kwargs)
    elif mode == "live":
        from engines.broker_qmt import MiniQMTBroker
        return MiniQMTBroker(**kwargs)
    elif mode == "backtest":
        from engines.backtest_engine import SimBroker, BacktestConfig
        config = kwargs.get("config", BacktestConfig())
        return SimBroker(config)
    else:
        raise ValueError(f"Unknown execution mode: {mode}")
```

```python
def calc_position_deviation(
    target_weights: dict[str, float],
    actual_shares: dict[str, int],
    prices: dict[str, float],
    total_value: float,
) -> float:
    """计算实际vs理论仓位偏差(百分比)。

    偏差 = mean(|actual_weight_i - target_weight_i|) * 100

    Args:
        target_weights: {code: target_weight} 目标权重。
        actual_shares: {code: shares} 实际持股。
        prices: {code: price} 当前价格。
        total_value: 组合总资产。

    Returns:
        平均绝对偏差(百分比)。空持仓返回0.0。
    """
    if not target_weights or total_value <= 0:
        return 0.0

    all_codes = set(target_weights.keys()) | set(actual_shares.keys())
    deviations = []
    for code in all_codes:
        target_w = target_weights.get(code, 0.0)
        actual_mv = actual_shares.get(code, 0) * prices.get(code, 0.0)
        actual_w = actual_mv / total_value if total_value > 0 else 0.0
        deviations.append(abs(actual_w - target_w))

    return float(np.mean(deviations) * 100) if deviations else 0.0
```

```python
def calc_vwap_bias(df: pd.DataFrame, window: int = 1) -> pd.Series:
    """VWAP偏差因子: (close - VWAP) / VWAP。

    VWAP = rolling_sum(amount*1000) / rolling_sum(volume*100)
    amount单位=千元, volume单位=手。

    Args:
        df: 含close/amount/volume列的DataFrame。
        window: VWAP计算窗口。

    Returns:
        VWAP偏差Series。
    """
    # 单位转换: amount千元→元, volume手→股
    total_amount = (df["amount"] * 1000).rolling(window).sum()
    total_volume = (df["volume"] * 100).rolling(window).sum()
    vwap = total_amount / total_volume
    vwap = vwap.replace([np.inf, -np.inf], np.nan)
    bias = (df["close"] - vwap) / vwap
    return bias.replace([np.inf, -np.inf], np.nan)


def calc_rsrs_raw(df: pd.DataFrame, window: int = 18) -> pd.Series:
    """RSRS原始斜率因子: rolling OLS(high ~ low)的斜率。

    Args:
        df: 含high/low列的DataFrame。
        window: 回归窗口。

    Returns:
        RSRS斜率Series。
    """
    from scipy import stats as sp_stats

    slopes = pd.Series(np.nan, index=df.index)
    highs = df["high"].values
    lows = df["low"].values

    for i in range(window - 1, len(df)):
        y = highs[i - window + 1: i + 1]
        x = lows[i - window + 1: i + 1]
        if np.any(np.isnan(x)) or np.any(np.isnan(y)):
            continue
        result = sp_stats.linregress(x, y)
        slopes.iloc[i] = result.slope

    return slopes
```
